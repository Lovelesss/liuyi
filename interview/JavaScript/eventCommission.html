# 事件委托

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title></title>
    <style>

    </style>
</head>
<body>
<ul id="wrap">
    <li>苹果</li>
    <li>橘子</li>
    <li>梨子</li>
</ul>
<script>
    const wrap = document.getElementById('wrap');
    wrap.addEventListener('click', (e) => {
        console.log(e.target.innerHTML);
    });
    let li4 = document.createElement('li');
    li4.innerHTML = '葡萄';
    wrap.appendChild(li4);
</script>
</body>
</html>
